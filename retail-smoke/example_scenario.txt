# знак "#" обозначает коментарий, его программа пропускает
# Описание команд:
#
# win - обозначает координаты верхнего левого угла браузера по которому записан сценарий, 
#данная команда обязательно должна быть первой в сценарии, параметры: x и y. Пример: win=-8,-8,1032,736
#
# lst - список из файлов сценариев в каталоге с программой для последовательного выполнения, параметры: имена файлов. 
#Пример: lst=prod.txt,fin.txt
#
# cl - одно нажатие левой кнопки мыши, параметры: x и y. Пример: cl=1000,800
#
# d_cl - двойное нажатие левой кнопки мыши, параметры: x и y. Пример: d_cl=1000,800
#
# in - ввести текст, параметры: произвольный текст за исключением символа "#". Пример: in=продукты
#
# pr - нажать кнопку, параметры: f4,space,down,enter,next_table,tab,exit,go_to_link,insert,paste,
#delete,select_all,bksp,cntrl_entr,esc. Пример: pr=cntrl_entr, delete
#
# sl - ожидать в течении указанного количества секунд. Пример: sl=3
#
#
win=-8,-8,1032,736
sl=2
cl=65,277 # Закупки
sl=2
cl=273,300 # Поставщики
sl=2
cl=778,251 # Поиск
in=Электро
sl=1
pr=enter # искать
